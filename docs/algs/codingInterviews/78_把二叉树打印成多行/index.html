      78. 把二叉树打印成多行
        • Description
        • Solutions
          • Level Order Traversal

      Description #

      给定一个节点数为 n 二叉树，要求从上到下按层打印二叉树的 val 值，同一层结点从左至右输出，每一层输出一行，将输出的结果存放到一个二维数组中返回。

      Solutions #

      Level Order Traversal #

      简单题，只需要进行一次层次遍历即可。

      func Print( pRoot *TreeNode ) [][]int {
    // write code here
    var res [][]int
    if pRoot == nil {
        return res
    }
    
    var queue []*TreeNode
    queue = append(queue, pRoot)
    
    for len(queue) != 0 {
        size := len(queue)
        var level []int
        for i := 0; i < size; i++ {
            node := queue[i]
            level = append(level, node.Val)
            if node.Left != nil {
                queue = append(queue, node.Left)
            }
            if node.Right != nil {
                queue = append(queue, node.Right)
            }
        }
        
        res = append(res, level)
        queue = queue[size:]
    }
    
    return res
}
